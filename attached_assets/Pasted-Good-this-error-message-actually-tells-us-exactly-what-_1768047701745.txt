```js
  const rawQuery = req.query.q;

  if (!rawQuery || rawQuery.trim() === "") {
    return res.send("<h2>Please enter a search term.</h2>");
  }

  const query = rawQuery.toLowerCase().trim();
  const posts = JSON.parse(fs.readFileSync("database.json"));

  const results = posts.filter(post =>
    post.status === "published" &&
    (
      post.title.toLowerCase().includes(query) ||
      post.content.toLowerCase().includes(query)
    )
  );

  let html = `<h2>Search results for "${rawQuery}"</h2>`;

  if (results.length === 0) {
    html += `<p>No articles found matching your search.</p>`;
  }

  results.forEach(post => {
    html += `
      <div style="margin-bottom:20px;">
        <h3>${post.title}</h3>
        <small>${formatDateTime(post.createdAt)}</small>
        <p>${post.content.replace(/<[^>]*>/g,"").substring(0,120)}...</p>
        <a href="/post/${post.id}">Read more</a>
      </div>
      <hr>
    `;
  });

  res.send(html);
});
```
