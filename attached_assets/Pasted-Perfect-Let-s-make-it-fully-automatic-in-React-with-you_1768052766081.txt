```ts
export function generateSitemap(posts: any[]) {
  const baseUrl = "https://yourwebsite.com";

  const header = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">`;

  const footer = "</urlset>";

  // Homepage
  const urls = [
    `
  <url>
    <loc>${baseUrl}/</loc>
    <lastmod>${new Date().toISOString().split("T")[0]}</lastmod>
    <changefreq>daily</changefreq>
    <priority>1.0</priority>
  </url>`
  ];

  // Categories
  const categories = [
    "market-news",
    "cryptocurrency",
    "personal-finance",
    "ipo-analysis",
    "stock-ideas",
    "fpo-analysis",
    "results"
  ];

  categories.forEach(cat => {
    urls.push(`
  <url>
    <loc>${baseUrl}/category/${cat}</loc>
    <lastmod>${new Date().toISOString().split("T")[0]}</lastmod>
    <changefreq>weekly</changefreq>
    <priority>0.9</priority>
  </url>`);
  });

  // All posts
  posts.forEach(post => {
    urls.push(`
  <url>
    <loc>${baseUrl}/post/${post.id}</loc>
    <lastmod>${new Date(post.createdAt).toISOString().split("T")[0]}</lastmod>
    <changefreq>weekly</changefreq>
    <priority>0.7</priority>
  </url>`);
  });

  return header + urls.join("") + footer;
}
```

‚úÖ This function returns the **full sitemap.xml content** as a string.

# üß© STEP 2: Update Admin Publish Function

In `adminCMS.tsx`:

```ts
import { generateSitemap } from "./utils/generateSitemap";

const handlePublish = () => {
  const existingPosts = JSON.parse(localStorage.getItem("posts") || "[]");

  const newPost = {
    id: Date.now(),
    title,
    content,
    category,
    createdAt: new Date().toISOString()
  };

  existingPosts.push(newPost);
  localStorage.setItem("posts", JSON.stringify(existingPosts));

  // Generate sitemap
  const sitemap = generateSitemap(existingPosts);
  localStorage.setItem("sitemap", sitemap); // store sitemap in localStorage for now

  alert("Post published & sitemap updated!");
};
```

```tsx
const downloadSitemap = () => {
  const sitemap = localStorage.getItem("sitemap") || "";
  const blob = new Blob([sitemap], { type: "application/xml" });
  const url = URL.createObjectURL(blob);

  const link = document.createElement("a");
  link.href = url;
  link.download = "sitemap.xml";
  link.click();
};
```

* Every time you publish, click ‚ÄúDownload Sitemap‚Äù ‚Üí upload to Replit root `/sitemap.xml`
