## ‚úÖ STEP 1: UPDATE YOUR FULL POST ROUTE

### üîπ Replace your existing `/post/:id` route with this

```js
app.get("/post/:id", (req, res) => {
  const posts = JSON.parse(fs.readFileSync("database.json"));

  const post = posts.find(p => p.id == req.params.id);
  if (!post) return res.send("Post not found");

  // Other posts (exclude current one)
  const otherPosts = posts
    .filter(p => p.id != post.id && p.status === "published")
    .slice(-5)
    .reverse();

  let html = `
    <html>
    <head>
      <title>${post.title}</title>
      <style>
        body {
          font-family: Arial, sans-serif;
          background: #f9f9f9;
          padding: 20px;
        }
        .post-container {
          background: #fff;
          padding: 25px;
          border-radius: 8px;
        }
        .post-meta {
          color: #666;
          font-size: 14px;
          margin-bottom: 15px;
        }
        .more-posts {
          margin-top: 40px;
        }
        .more-posts h2 {
          color: #137333;
        }
        .post-card {
          background: #fff;
          padding: 15px;
          margin-bottom: 15px;
          border-radius: 6px;
          box-shadow: 0 2px 6px rgba(0,0,0,0.1);
        }
        .post-card h3 {
          margin: 0 0 8px;
        }
        .post-card small {
          color: #777;
        }
        .post-card a {
          color: #137333;
          text-decoration: none;
          font-weight: 600;
        }
      </style>
    </head>
    <body>

      <div class="post-container">
        <h1>${post.title}</h1>

        <div class="post-meta">
          ${formatDateTime(post.createdAt)} | ${post.category.replace(/-/g, " ")}
        </div>

        <div class="post-content">
          ${post.content}
        </div>
      </div>

      <div class="more-posts">
        <h2>More Articles You May Like</h2>
  `;

  if (otherPosts.length === 0) {
    html += `<p>No more articles available.</p>`;
  }

  otherPosts.forEach(p => {
    html += `
      <div class="post-card">
        <h3>${p.title}</h3>
        <small>${formatDateTime(p.createdAt)}</small>
        <p>${p.content.replace(/<[^>]*>/g, "").substring(0, 120)}...</p>
        <a href="/post/${p.id}">Read Full Article ‚Üí</a>
      </div>
    `;
  });

  html += `
      </div>
    </body>
    </html>
  `;

  res.send(html);
});
```
