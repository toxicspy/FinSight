# ‚úÖ STEP 2: ADD SEARCH ROUTE (BACKEND ‚Äì VERY IMPORTANT)

Paste this into **server.js** (above `app.listen`):

```js
app.get("/search", (req, res) => {
  const query = req.query.q?.toLowerCase();

  if (!query) {
    return res.send("<h2>Please enter a search term.</h2>");
  }

  const posts = JSON.parse(fs.readFileSync("database.json"));

  const results = posts.filter(post =>
    post.status === "published" &&
    (
      post.title.toLowerCase().includes(query) ||
      post.content.toLowerCase().includes(query)
    )
  );

  let html = `
    <html>
    <head>
      <title>Search results for "${query}"</title>
      <style>
        body { font-family: Arial; background:#f9f9f9; padding:20px; }
        .result { background:#fff; padding:15px; margin-bottom:15px; border-radius:6px; }
        .result h3 { margin:0 0 6px; }
        .result small { color:#666; }
        .result a { color:#137333; font-weight:600; text-decoration:none; }
      </style>
    </head>
    <body>

    <h2>Search Results for "${query}"</h2>
  `;

  if (results.length === 0) {
    html += `<p>No matching articles found.</p>`;
  }

  results.forEach(post => {
    html += `
      <div class="result">
        <h3>${post.title}</h3>
        <small>${formatDateTime(post.createdAt)} | ${post.category.replace(/-/g," ")}</small>
        <p>${post.content.replace(/<[^>]*>/g,"").substring(0,150)}...</p>
        <a href="/post/${post.id}">Read Full Article ‚Üí</a>
      </div>
    `;
  });

  html += `
    </body>
    </html>
  `;

  res.send(html);
});
```
